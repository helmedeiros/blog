{{/* Series navigation. Renders an ordered list of every post that shares the current page's series, ordered by `series_order`, with the current post highlighted. Only renders when the post has a `series` param and at least one other post in the same series exists. Bilingual (EN/PT) based on .Language.Lang. */}} {{- if .Params.series -}} {{- $currentSeries := index .Params.series 0 -}} {{- $currentPermalink := .Permalink -}} {{- $seriesPages := where (where .Site.RegularPages "Section" "posts") "Params.series" "intersect" (slice $currentSeries) -}} {{- $seriesPages := $seriesPages.ByParam "series_order" -}} {{- if gt (len $seriesPages) 1 -}} {{- $isPT := eq .Language.Lang "pt" -}} {{- $title := cond $isPT "Outros posts dessa série" "More in this series" -}} {{- $hereLabel := cond $isPT "você está lendo este" "you're reading this one" -}}

{{ $title }}

    {{- range $seriesPages }}
  1. {{- if eq .Permalink $currentPermalink }} {{ .Title }} — {{ $hereLabel }} {{- else }} {{ .Title }} {{- end }}
    2. {{- end }}
{{- end -}} {{- end -}}
