{{/* Series navigation. Renders an ordered list of every post that shares the current page's series, ordered by `series_order`, with the current post highlighted. Pulls the series title from `content/{lang}/series/{slug}.md` when it exists (matches the OKRA convention) so the heading reads naturally; falls back to the humanised slug otherwise. Bilingual (EN/PT) via `.Language.Lang`. Renders nothing on posts without a series. */}} {{- if .Params.series -}} {{- $currentSeries := index .Params.series 0 -}} {{- $currentPermalink := .Permalink -}} {{- $seriesPages := where (where .Site.RegularPages "Section" "posts") "Params.series" "intersect" (slice $currentSeries) -}} {{- $seriesPages := $seriesPages.ByParam "series_order" -}} {{- if gt (len $seriesPages) 1 -}} {{/* Try to find the series term page for a friendly title. Hugo exposes taxonomy term pages on .Site.GetPage. */}} {{- $termPath := printf "/series/%s" $currentSeries -}} {{- $term := .Site.GetPage $termPath -}} {{- $seriesTitle := "" -}} {{- if $term }}{{ $seriesTitle = $term.Title }}{{ end -}} {{- if not $seriesTitle }}{{ $seriesTitle = humanize $currentSeries }}{{ end -}} {{- $isPT := eq .Language.Lang "pt" -}} {{- $eyebrow := cond $isPT "Parte da série" "Part of the series" -}} {{- $hereLabel := cond $isPT "você está lendo este" "you're reading this one" -}} {{- $viewAll := cond $isPT "Ver todos os posts da série →" "View all posts in this series →" -}}

{{ $eyebrow }} {{ $seriesTitle }}

    {{- range $seriesPages }}
  1. {{- if eq .Permalink $currentPermalink }} {{ .Title }} — {{ $hereLabel }} {{- else }} {{ .Title }} {{- end }}
    2. {{- end }}

{{ $viewAll }}

{{- end -}} {{- end -}}
